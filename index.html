from fastapi import FastAPI from pydantic import BaseModel from transformers import AutoTokenizer, AutoModelForSequenceClassification import torch import numpy as np app = FastAPI() # FinBERTモデルのロード（金融テキストに最適化されたBERT） tokenizer = AutoTokenizer.from_pretrained("ProsusAI/finbert") model = AutoModelForSequenceClassification.from_pretrained("ProsusAI/finbert") class SentimentRequest(BaseModel): text: str source_type: str # 'central_bank', 'politician', 'news' impact_level: str # 'high', 'medium', 'low' def calculate_fsi(text): # 1. センチメント分析 inputs = tokenizer(text, return_tensors="pt", padding=True, truncation=True, max_length=512) outputs = model(**inputs) prediction = torch.nn.functional.softmax(outputs.logits, dim=-1) # FinBERT output: [positive, negative, neutral] probs = prediction.detach().numpy()[0] pos, neg, neu = probs[0], probs[1], probs[2] # 0-100のスコアに変換 (50が中立) # スコア = (ポジティブ確率 * 100 * 0.5) + (50) - (ネガティブ確率 * 100 * 0.5) base_score = 50 + (pos * 50) - (neg * 50) return int(base_score), float(pos), float(neg) @app.post("/analyze") async def analyze_sentiment(req: SentimentRequest): score, bullish_p, bearish_p = calculate_fsi(req.text) # 重み付け設定 weights = {"central_bank": 3, "politician": 2, "news": 1} impacts = {"high": 1.5, "medium": 1.0, "low": 0.5} weight = weights.get(req.source_type, 1) impact = impacts.get(req.impact_level, 1.0) # 最終的な重み付きスコア（単一入力の場合） final_score = score return { "fsi_score": final_score, "interpretation": "Bullish" if final_score > 60 else "Bearish" if final_score < 40 else "Neutral", "bullish_probability": round(bullish_p * 100, 2), "bearish_probability": round(bearish_p * 100, 2), "weight_applied": weight, "impact": req.impact_level } if __name__ == "__main__": import uvicorn uvicorn.run(app, host="0.0.0.0", port=8000)
